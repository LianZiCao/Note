#include<iostream>
#include<vector>
#include<set>
#include<limits.h>
#include<cctype>
#include<algorithm>
#include<utility>

using namespace std;


int main() {
	int n, c;
	cin >> n >> c;
	int count = 0;
	multiset<int> s;
	while (n--) {
		int num;
		cin >> num;
		count += s.count(num + c) + s.count(num - c);
		s.insert(num);
	}
	cout << count;
}
