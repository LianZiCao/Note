#include<iostream>
#include<vector>
#include<algorithm>
#include<iomanip>
#include<cmath>
#include<numeric>

using namespace std;

vector<int> prefix_function(string s) {
	int n = s.size();
	vector<int> pi(n);
	for (int i = 1; i < n; i++) {
		int j = pi[i - 1];
		while (j > 0 && s[i] != s[j])j = pi[j - 1];
		if (s[i] == s[j]) j++;
		pi[i] = j;
	}
	return pi;
}

vector<int> happen(string text, string pattern) {
	string cur = pattern + "#" + text;
	int size1 = text.size();
	int size2 = pattern.size();
	vector<int> v;
	vector<int> lps = prefix_function(cur);
	for (int i = size2 + 1; i < size1 + size2 + 1; i++) {
		if (lps[i] == size2)
			v.push_back(i - 2 * size2);
	}
	return v;
}

int main() {
	string s1, s2;
	cin >> s1 >> s2;
	vector<int> occur = happen(s1, s2);
	vector<int> prefix = prefix_function(s2);
	for (auto& x : occur) {
		cout << x + 1 << endl;
	}
	for (auto& x : prefix) {
		cout << x << ' ';
	}
}
