问gpt搜了篇关于单调栈的文章再粗读了一下没思路问gpt的，然后自己又写了一次。要注意的是这里**要存的是下标**，因为输出是下标，对下标进行存储，最后出栈的时候处理，并且也是很经典的检测`!s.empty()`防止访问空栈的顶部

这边维护的栈是单调递减栈，**当出现比栈顶大的元素时，就说明比栈顶大的元素是比较的数字**，而`i + 1`就是转换为以1为开始。或者说，`v[s.top()] < v[i]`这串表达式即对应题目的第一个**大于**$a_i$的元素的下标


AC
```c++
#include<iostream>
#include<vector>
#include<algorithm>
#include<iomanip>
#include<cmath>
#include<stack>

using namespace std;

int main() {

    int n;
    cin >> n;
    vector<int> v(n);
    for (auto& x : v)
        cin >> x;

    stack<int> s;
    vector<int> ans(n, 0);
    for (int i = 0; i < n; i++) {
        while (!s.empty() && v[s.top()] < v[i]) {
            ans[s.top()] = i + 1;
            s.pop();

        }
        s.push(i);
    }
    for (auto& x : ans)
        cout << x << ' ';

    return 0;
}
```
